291 template<typename Item>
292 bool PipeLine<Item>::constructPipelinesByScenario()
293 {
294  std::unique_lock<std::mutex> lk(m_mutex);
295  if (!checkValid()) return false;
296  if (!m_pipelineModified) return true;
297 
298  for (auto[notifierType, notifiers] : m_notifierMaps) {
299  for (auto notifier : notifiers) {
300  m_pipeNodeDispatcher->addNotifier(notifier);
301  }
302  }
303 
304  //get all scenario
305  for (auto&[nodeId, node] : m_pipeNodeMaps) {
306  for (PipelineScenario scenario : node->getScenarios()) {
307  m_pipelineScenarioSet.insert(scenario);
308  }
309  }
310  auto pipelines = m_dag.multiTopologicalSort();
311 
312  for (auto it = m_pipelineScenarioSet.begin(); it != m_pipelineScenarioSet.end(); it++) {
313  PipelineScenario scenario = *it;
314  VTF_LOGD("scenario {0} ", scenario);
315  for (auto pipelineIter = pipelines.begin(); pipelineIter != pipelines.end(); pipelineIter++) {
316  auto pipeline = *pipelineIter;
317  bool findFlag = true;
318  for (long &nodeID : pipeline) {
319  if (m_pipeNodeMaps.count(nodeID) == 0) {
320  VTF_LOGE("can't find node {0} in m_pipeNodeMaps, please check it.", nodeID);
321  m_scenario2PipelineMaps.clear();
322  return false;
323  }
324  if (!m_pipeNodeMaps[nodeID]->hasScenario((PipelineScenario)scenario)) {
325  findFlag = false;
326  break;
327  }
328  }
329  if (findFlag) {
330  if (m_scenario2PipelineMaps.count((PipelineScenario)scenario)) {
331  VTF_LOGE("There can only be one pipeline in a scenario {0}, please check Graph.", scenario);
332  m_scenario2PipelineMaps.clear();
333  return false;
334  }
335  m_scenario2PipelineMaps[(PipelineScenario)scenario] = pipeline;
336  }
337  }
338  }
339 
340  m_pipelineModified = false;
341  dumpPipelines();
342  return true;
343 }
