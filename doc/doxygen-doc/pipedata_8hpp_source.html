221 bool PipeData::constructDependency(const std::vector<long>& pipeline)
222 {
223  m_dependencies.clear();
224  for (int i = 0; i < pipeline.size(); i++) {
225  long curNodeId = pipeline[i];
226 
227  //check dependency exist
228  auto it = std::find_if(m_dependencies.begin(), m_dependencies.end(), [curNodeId](const Dependency& dependency) {
229  return dependency.curNodeId == curNodeId;
230  });
231  if (it != m_dependencies.end()) {
232  VTF_LOGE("already exist dependency of node {0}, please check it first.", curNodeId);
233  m_dependencies.clear();
234  }
235 
236  Dependency dependency = Dependency{.curNodeId = curNodeId};
237  if (i - 1 >= 0) {
238  dependency.precursors = std::make_pair<>(pipeline[i-1], DependencyStatus::NOREADY);
239  } else {
240  dependency.precursors = std::make_pair<>(-1, DependencyStatus::DONE);
241  }
242  if (i + 1 < pipeline.size()) {
243  dependency.successors = std::make_pair<>(pipeline[i+1], DependencyStatus::NOREADY);
244  } else {
245  dependency.successors = std::make_pair<>(-1, DependencyStatus::DONE);
246  }
247  m_dependencies.push_back(dependency);
248  }
249 
250  if (m_dependencies.empty() || m_dependencies.size() <= 1) {
251  VTF_LOGE("dependency size of pipe request can't be less than 2");
252  return false;
253  }
254 
255  //set current process node to first node
256  m_currentProcessNodeId = m_dependencies[0].curNodeId;
257  m_currentProcessNodeIdx = 0;
258  m_nextNodeIdx = m_currentProcessNodeIdx + 1;
259  m_nextNodeId = findNextNode();
260 
261 
262  //dump
263  if (m_enableDebug) {
264  for (auto& dependency : m_dependencies) {
265  VTF_LOGD("node : [{0}]", dependency.curNodeId);
266  VTF_LOGD("pre : [{0}]", dependency.precursors.first);
267  VTF_LOGD("suc : [{0}]", dependency.successors.first);
268  VTF_LOGD("--------------------------------------");
269  }
270  }
271 
272  return true;
273 }

275 long PipeData::findNextNode()
276 {
277  if (!checkDependencyValid()) return -1;
278  Dependency currentDependency = m_dependencies[m_currentProcessNodeIdx];
279  long successorId = currentDependency.successors.first;
280  DependencyStatus successorStatus = currentDependency.successors.second;
281  if (successorId != -1 && successorStatus == DependencyStatus::NOREADY) {
282  return successorId;
283  }
284  return -1;

287 bool PipeData::checkDependencyIsReady()
288 {
289  if (!checkDependencyValid()) return false;

291  long precursorId = currentDependency.precursors.first;

293

295  || (precursorId != -1 && precursorStatus == DependencyStatus::READY))
296  {

304 void PipeData::markCurrentNodeReady()
305 {
306  long nextNodeId = findNextNode();
307  m_nextNodeIdx = m_currentProcessNodeIdx + 1;
308  //last node
309  if (nextNodeId == -1 || m_nextNodeIdx >= m_dependencies.size()) {
310  VTF_LOGD("request {0} node [{1}] have done.", ID(), m_currentProcessNodeId);
311  m_currentProcessNodeId = -1;
312  m_currentProcessNodeIdx++;
313  m_nextNodeId = -1;
314  m_nextNodeIdx = -1;
315  VTF_LOGD("request {0} all node already process done.", ID());
316  return;
317  }
318 
319  if (m_dependencies[m_nextNodeIdx].precursors.first != m_dependencies[m_currentProcessNodeIdx].curNodeId) {
320  VTF_LOGD("node [{0}] and node [{1}] no connection. please check dependency.",
321  m_dependencies[m_currentProcessNodeIdx].curNodeId,
322  m_dependencies[m_nextNodeIdx].precursors.first
323  );
324  return;
325  }
326 
327  //mark next node denpendency's pre is done
328  m_dependencies[m_nextNodeIdx].precursors.second = DependencyStatus::READY;
329  VTF_LOGD("request {0} node [{1}] have done. ", ID(), m_currentProcessNodeId);
330 
331  m_currentProcessNodeId = nextNodeId;
332  m_currentProcessNodeIdx++;
333  m_nextNodeId = findNextNode();
