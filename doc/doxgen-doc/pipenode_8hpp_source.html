203 template<typename Item>
204 bool PipeNode<Item>::process(std::shared_ptr<Item> item)
205 {
206  bool ret = true;
207  m_status = PipeNodeStatus::PROCESSING;
208  ret = m_processCallback(item);
209  if (ret) {
210  item->markCurrentNodeReady();
211  auto pipeNodeDispatcherSp = m_pipeNodeDispatcher.lock();
212  if (pipeNodeDispatcherSp) {
213  pipeNodeDispatcherSp->queueInDispacther(item);
214  pipeNodeDispatcherSp->notifyNotFinal(item);
215  }
216  }
217  m_status = PipeNodeStatus::IDLE;
218  return ret;
219 }
