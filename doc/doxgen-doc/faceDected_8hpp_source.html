111 bool dnnfacedetect::detect(std::shared_ptr<FrameRequest> request)
112 {
113  std::unique_lock<std::mutex> lk(m_mutex);
114 
115  VTF_LOGD("faceDected inScaleFactor = {0} this {1}", inScaleFactor, fmt::ptr(this));
116  auto start = std::chrono::system_clock::now();
117  Mat tmpsrc = *(request->getFrame());
118  // 修改通道数
119  if (tmpsrc.channels() == 4)
120  cvtColor(tmpsrc, tmpsrc, COLOR_BGRA2BGR);
121  // 输入数据调整
122  Mat detection;
123  {
124  Mat inputBlob = dnn::blobFromImage(tmpsrc, inScaleFactor,
125  Size(inWidth, inHeight), meanVal, false, false);
126  //人脸检测
127  _net.setInput(inputBlob, "data");
128  detection = _net.forward("detection_out");
129  }
130 
131 
132 
133  Mat detectionMat(detection.size[2], detection.size[3],
134  CV_32F, detection.ptr<float>());
135  VTF_LOGD("faceDected dected {0} face, inScaleFactor = {1} this {2}", detectionMat.rows, inScaleFactor, fmt::ptr(this));
136  //检测出的结果进行绘制和存放到dsts中
137  for (int i = 0; i < detectionMat.rows; i++)
138  {
139  //置值度获取
140  float confidence = detectionMat.at<float>(i, 2);
141  //如果大于阈值说明检测到人脸
142 
143  if (confidence > confidenceThreshold)
144  {
145  //计算矩形
146  int xLeftBottom = static_cast<int>(detectionMat.at<float>(i, 3) * tmpsrc.cols);
147  int yLeftBottom = static_cast<int>(detectionMat.at<float>(i, 4) * tmpsrc.rows);
148  int xRightTop = static_cast<int>(detectionMat.at<float>(i, 5) * tmpsrc.cols);
149  int yRightTop = static_cast<int>(detectionMat.at<float>(i, 6) * tmpsrc.rows);
150  //生成矩形
151  Rect rect((int)xLeftBottom, (int)yLeftBottom,
152  (int)(xRightTop - xLeftBottom),
153  (int)(yRightTop - yLeftBottom));
154 
155  //在原图上用红框画出矩形
156  rectangle(*(request->getFrame()), rect, Scalar(0, 0, 255));
157  }
158  }
159  auto end = std::chrono::system_clock::now();
160  std::chrono::duration<double> runTime = end - start;
161  VTF_LOGD("faceDected id{0} need {1}ms ", request->ID(), vtf::utils::TimeUtil::convertTime<std::chrono::milliseconds>(runTime).count());
162  return true;
