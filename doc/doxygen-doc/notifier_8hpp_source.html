159 template<typename Item>
160 bool Notifier<Item>::threadLoop(std::shared_ptr<Item> item)
161 {
162  bool ret = true;
163  if (!m_processCallback) {
164  VTF_LOGD("user must give a process function for notify.");
165  return false;
166  }
167  if (item->ID() == m_expectItemId) {
168  //if current item id equal m_expectItemId
169 
170  //process current item first
171  ret = m_processCallback(item);
172  //update m_expectItemId
173  m_expectItemId = item->ID() + 1;
174  VTF_LOGD("[{0}] result notifier process item {1}, now expect item id {2}", name(), item->ID(), m_expectItemId);
175 
176  //if pending map is not empty, mean future item arrive first, we need process it.
177  for (auto it = m_pendingItemMap.begin(); it != m_pendingItemMap.end();) {
178  //need one-by-one process, so we need juge m_expectItemId and item id.
179  if (it->first == m_expectItemId) {
180  m_processCallback(it->second);
181  //update m_expectItemId
182  m_expectItemId = it->second->ID() + 1;
183  VTF_LOGD("[{0}] result notifier process item {1}, now expect item id {2}", name(), it->second->ID(), m_expectItemId);
184  m_pendingItemMap.erase(it++);
185  } else {
186  it++;
187  break;
188  }
189  }
190  } else {
191  //if current item not equal m_expectItemId, will hold it to pending map
192  m_pendingItemMap[item->ID()] = item;
193  }
194  return ret;
195 }
