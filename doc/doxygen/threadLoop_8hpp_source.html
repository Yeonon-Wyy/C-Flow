86 void ThreadLoop<T>::_threadLoop()
87 {
88  VTF_LOGD("ThreadLoop start");
89  while (true) {
90  T item;
91  {
92  std::unique_lock<std::mutex> lk(m_mutex);
93  m_condition.wait(lk, [this]() {
94  return this->m_isStop || !m_itemQueue.empty();
95  });
96 
97  //if isStop flag was set and m_itemQueue is empty, exit loop
98  //so, if isStop flag was set, but m_itemQueue is not empty, the loop will execute continue untile queue is empty
99  if (this->m_isStop && m_itemQueue.empty()) {
100  break;
101  }
102  item = m_itemQueue.front();
103  m_itemQueue.pop();
104 
105  //if m_itemQueue'size less than m_queueSize, notify it.
106  if (m_itemQueue.size() < m_queueSize) {
107  m_not_full_cv.notify_one();
108  }
109  }
110 
111  bool result = this->threadLoop(item);
112  if (!result) {
113  return;
114  }
115  }
