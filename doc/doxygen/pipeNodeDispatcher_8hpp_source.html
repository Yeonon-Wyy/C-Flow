124 template<typename Item>
125 bool PipeNodeDispatcher<Item>::dispatch(std::shared_ptr<Item> item)
126 {
127  VTF_LOGD("dispatch item id({0}) nextNodeId {1}", item->ID(), item->getCurrentNode());
128  //final or pipeline is stoped, will call notifier
129  if (item->getCurrentNode() == -1 || m_threadPool.isStoped()) {
130  //just call final notifier
131  if (item->getCurrentNode() == -1) {
132  notifyFinal(item, NotifyStatus::OK);
133  } else {
134  notifyFinal(item, NotifyStatus::ERROR);
135  }
136  return true;
137  }
138 
139  if (item->checkDependencyIsReady()) {
140  long currentProcessNodeId = item->getCurrentNode();
141  if (currentProcessNodeId < 0) {
142  //finish
143  return true;
144  }
145  if (m_pipeNodeMaps.count(currentProcessNodeId) == 0) {
146  VTF_LOGE("can't find currentNode node {0}", currentProcessNodeId);
147  return true;
148  }
149 
150  auto currentNodeSp = m_pipeNodeMaps[currentProcessNodeId].lock();
151  //submit to thread pool
152  if (currentNodeSp && !m_threadPool.isStoped() && !currentNodeSp->isStop()) {
153  currentNodeSp->submit(item);
154  } else {
155  //if node already stop or destory, should notify error
156  notifyFinal(item, NotifyStatus::ERROR);
157  }
158  }
159  return true;
160 }
