288 template<typename Item>
289 bool PipeLine<Item>::constructPipelinesByScenario()
290 {
291  std::unique_lock<std::mutex> lk(m_mutex);
292  if (!checkValid()) return false;
293  if (!m_pipelineModified) return true;
294 
295  for (auto[notifierType, notifiers] : m_notifierMaps) {
296  for (auto notifier : notifiers) {
297  m_pipeNodeDispatcher->addNotifier(notifier);
298  }
299  }
300 
301  //get all scenario
302  for (auto&[nodeId, node] : m_pipeNodeMaps) {
303  for (PipelineScenario scenario : node->getScenarios()) {
304  m_pipelineScenarioSet.insert(scenario);
305  }
306  }
307  auto pipelines = m_dag.multiTopologicalSort();
308 
309  for (auto it = m_pipelineScenarioSet.begin(); it != m_pipelineScenarioSet.end(); it++) {
310  PipelineScenario scenario = *it;
311  VTF_LOGD("scenario {0} ", scenario);
312  for (auto pipelineIter = pipelines.begin(); pipelineIter != pipelines.end(); pipelineIter++) {
313  auto pipeline = *pipelineIter;
314  bool findFlag = true;
315  for (long &nodeID : pipeline) {
316  if (m_pipeNodeMaps.count(nodeID) == 0) {
317  VTF_LOGE("can't find node {0} in m_pipeNodeMaps, please check it.", nodeID);
318  m_scenario2PipelineMaps.clear();
319  return false;
320  }
321  if (!m_pipeNodeMaps[nodeID]->hasScenario((PipelineScenario)scenario)) {
322  findFlag = false;
323  break;
324  }
325  }
326  if (findFlag) {
327  if (m_scenario2PipelineMaps.count((PipelineScenario)scenario)) {
328  VTF_LOGE("There can only be one pipeline in a scenario {0}, please check Graph.", scenario);
329  m_scenario2PipelineMaps.clear();
330  return false;
331  }
332  m_scenario2PipelineMaps[(PipelineScenario)scenario] = pipeline;
333  }
334  }
335  }
336 
337  m_pipelineModified = false;
338  dumpPipelines();
339  return true;
340 }
