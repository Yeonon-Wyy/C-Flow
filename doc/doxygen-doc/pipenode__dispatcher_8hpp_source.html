121 template<typename Item>
122 bool PipeNodeDispatcher<Item>::dispatch(std::shared_ptr<Item> item)
123 {
124  VTF_LOGD("dispatch item id({0}) nextNodeId {1}", item->ID(), item->getCurrentNode());
125  //final or pipeline is stoped, will call notifier
126  if (item->getCurrentNode() == -1 || m_threadPool.isStoped()) {
127  //just call final notifier
128  if (item->getCurrentNode() == -1) {
129  notifyFinal(item, NotifyStatus::OK);
130  } else {
131  notifyFinal(item, NotifyStatus::ERROR);
132  }
133  return true;
134  }
135 
136  if (item->checkDependencyIsReady()) {
137  long currentProcessNodeId = item->getCurrentNode();
138  if (currentProcessNodeId < 0) {
139  //finish
140  return true;
141  }
142  if (m_pipeNodeMaps.count(currentProcessNodeId) == 0) {
143  VTF_LOGE("can't find currentNode node {0}", currentProcessNodeId);
144  return true;
145  }
146 
147  auto currentNodeSp = m_pipeNodeMaps[currentProcessNodeId].lock();
148  //submit to thread pool
149  if (currentNodeSp && !m_threadPool.isStoped() && !currentNodeSp->isStop()) {
150  m_threadPool.emplace(&PipeNode<Item>::process, currentNodeSp, item);
151  // currentNodeSp->submit(item);
152  } else {
153  //if node already stop or destory, should notify error
154  notifyFinal(item, NotifyStatus::ERROR);
155  }
156  }
157  return true;
158 }
